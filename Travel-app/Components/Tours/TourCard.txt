import React, { useState } from "react";
import { Card, CardContent, CardFooter } from "@/components/ui/card";
import { Button } from "@/components/ui/button";
import { Badge } from "@/components/ui/badge";
import { Calendar, Clock, TrendingUp } from "lucide-react";
import BookingModal from "../booking/BookingModal";

export default function TourCard({ tour }) {
  const [showBooking, setShowBooking] = useState(false);

  return (
    <>
      <Card className="group overflow-hidden hover:shadow-xl transition-all duration-300 border-0 shadow-md flex flex-col">
        <div className="relative h-48 overflow-hidden">
          <img 
            src={tour.image_url} 
            alt={tour.name}
            className="w-full h-full object-cover group-hover:scale-110 transition-transform duration-500"
          />
        </div>
        <CardContent className="p-4 flex-grow">
          <Badge variant="secondary" className="mb-2">{tour.destination}</Badge>
          <h3 className="font-bold text-lg text-slate-800 mb-2">{tour.name}</h3>
          <div className="flex items-center text-sm text-slate-500 space-x-4 mb-3">
            <div className="flex items-center gap-1">
              <Clock className="w-4 h-4" />
              <span>{tour.duration_days} days</span>
            </div>
            <div className="flex items-center gap-1">
              <TrendingUp className="w-4 h-4" />
              <span>{tour.difficulty_level}</span>
            </div>
          </div>
        </CardContent>
        <CardFooter className="p-4 flex justify-between items-center bg-slate-50">
          <div>
            <p className="text-xl font-bold text-slate-800">${tour.price}</p>
            <p className="text-xs text-slate-500">per person</p>
          </div>
          <Button 
            onClick={() => setShowBooking(true)}
            className="bg-orange-500 hover:bg-orange-600"
            disabled={tour.available_spots === 0}
          >
            {tour.available_spots === 0 ? 'Sold Out' : 'Book Now'}
          </Button>
        </CardFooter>
      </Card>
      <BookingModal
        isOpen={showBooking}
        onClose={() => setShowBooking(false)}
        item={tour}
        bookingType="tour"
      />
    </>
  );
}
